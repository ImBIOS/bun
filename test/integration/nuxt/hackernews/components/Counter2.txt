<template>
  <div id="counter-fixture" class="rounded-br-full">
    <p>Count B: {{ count }}</p>
    <button class="inc" @click="increment">
      +
    </button>
    <button class="dec" @click="decrement">
      -
    </button>
  </div>
</template>

<script>
export default {
  data() {
    return {
      count: 0,
    };
  },
  mounted() {
    console.log('counter b loaded');
  },
  methods: {
    increment() {
      this.count += 2;
    },
    decrement() {
      this.count -= 2;
    },
  },
};
</script>
